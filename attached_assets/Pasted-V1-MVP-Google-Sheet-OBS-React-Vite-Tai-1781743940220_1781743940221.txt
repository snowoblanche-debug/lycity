權重抽卡
function weightedRandom(cards) {
  const totalWeight = cards.reduce(
    (sum, card) => sum + card.weight,
    0
  )

  let random =
    Math.random() * totalWeight

  for (const card of cards) {

    random -= card.weight

    if (random <= 0) {
      return card
    }
  }
}
